\recette{Sauce BBQ}
\preptime{5 min} \cooktime{20 min} \people{2} \robot{}

\recipe{
        \unit[1]{c-à-s} & Huile Végétale\\
        1 & Oignon Jaune -- (en \unit[90]{gr})\\
        \unit[3]{c-à-c} & Paprika Fumé \\
        \unit[3]{pincée} & Piment \\
        \unit[250]{gr} & Tomates Pelées\\
        \unit[1]{c-à-s} & Vinaigre de Cidre\\
        \unit[60]{ml} & Sirop D'Érable\\
        \unit[1]{c-à-s}& Vinaigre Balsamique \\
        \unit[1]{c-à-s}& Sauce sauja\\

}{
    \item Émincer l'oignon grossierement et le faire revenir a feux moyen avec l'huile
    \item Émincer l'ail et l'ajouter à l'oignon et laisser revenir.
    \item Séparement, préparer une mixture en mélangeant les épices, vinaigres et sirop d'érable.
    \item Laisser mijoter a feu doux une quinzaine de minute.
    \item Une fois la cuisson terminée, mixer le tout.
}
