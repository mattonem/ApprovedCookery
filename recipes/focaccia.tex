\recette{Focaccia}
\preptime{30 min} \cooktime{1 h 30} \people{4} \robot{}

\recipe{
        
        \unit[250]{gr} & Farine à Pain (T65)\\
        \unit[1]{sachet} & Levure Déshydratée\\
        \unit[2]{c-à-s} & Sel \\
        \unit[1]{c-à-c} & Sucre \\
        \unit[150]{ml} & Eau Tiède \\
        \unit[2]{c-à-s} & Huile d'Olive\\
    
}{
    \item Melanger l'eau, la levure, le sel, le sucre et l'huile et laisser la levure reagir quelque minutes. 
    \item Incorporer la mixture dans la farine et petrir jusqu'a l'obtention d'une pâte lisse. 
    \item Huiler les parois d'un grand saladier.
    \item Déposer la pâte dans dans le saladier et couvrir d'un torchon mouillé. 
    \item Laisser lever environ 1h (elle doit doubler de volume). 
    \item Étaler la pâte grossierment a la main en l'étirant. 
    \item Déposer la garniture (cristaux de sel, oignons, huile d'olive, etc...), lasser lever ~10 min
    \item Préchauffer le four (220 \textdegree C) en laissant un recient d'eau a l'interieur. 
    \item Enfourner la fougasse.
}
