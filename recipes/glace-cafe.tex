\recette{Glace au café}
\preptime{long} \people{\unit[500]{gr}} \robot{Sobetière}

\recipe{
        \unit[110]{gr} & Sucre\\
        \unit[100]{ml} & Expresso froid\\
        \unit[25]{cl} & Crème fraîche \\
        3 & Jaune d'{\oe}uf \\
        1 & Blanc d'{\oe}uf \\
}{
    \item Mettre la sorbetière au congélteur la veille.
    \item Battre la crème fraiche en chantilly bien solide et y incorporer un tier du sucre.
    \item Battre les jaunes d'{\oe}uf avec un tier du sucre jusqu'à se qu'ils blanchissent.
    \item Battre les blancs d'{\oe}uf en neige et y incorporer le reste du sucre.
    \item Placer les trois mixture et le café au réfrégirateur  $\sim$1h.
    \item Quand toutes le preparations sont bien froides, delicatement incorporer le un tier du café dans chacune d'entre elle puis les incorporer toutes ensembles (un travail grossier suffit car la sorbetière homogénéisera la mixture).
     \item Placer notre mixture dans la sorbetière et laisser tourner au moins 30min.
    \item Mettre au congélateur $\sim$4h. 
}
