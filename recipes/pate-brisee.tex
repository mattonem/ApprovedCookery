\recette{Pâte brisée}
\preptime{10 min} 
%100gr beurre
%300gr farine
%Sauce
%Sel
%8cl eau tiède
\recipe{
        \unit[100]{gr} & Beurre\\
        \unit[300]{gr} & Farine\\
        \unit[8]{cl} & Eau Tiède\\
        & Sauge \\
        & Sel \\
        }{
    \item Mélanger la farine, le sel et la sauge dans un plat.
    \item Ajouter le beurre puis l'incorporer à la farine en pétrissant rapidement et légèrement. On doit obtenir une sorte de semoule grossière.
    \item Incorporer rapidement l'eau. Continuer de pétrir jusqu'à obtention d'une pâte lisse et homogène.

}
